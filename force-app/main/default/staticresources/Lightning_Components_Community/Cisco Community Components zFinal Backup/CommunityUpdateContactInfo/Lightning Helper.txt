    getContactInfo : function(component) {
        var action = component.get("c.getContactDetail");
        action.setCallback(this, function(response) {
            var state = response.getState();
            if (state === "SUCCESS") {
                console.log('Response from Contact:'+response.getReturnValue());
                component.set("v.contactList",response.getReturnValue());
                var contactWrap = JSON.parse(JSON.stringify(component.get("v.contactList")));
                console.log('Contact WRAP PIYUSH :',contactWrap[0].conCountryAreaCode);
                if(contactWrap[0].conCountryAreaCode != null)
                {
                   component.set("v.selectedAreaCode","+"+contactWrap[0].conCountryAreaCode) ;
                }
                else
                {
                    component.set("v.selectedAreaCode","Country Code");
                }
                
            }
            else if (state === "INCOMPLETE") {
                // do something
            }
            else if (state === "ERROR") {
                var errors = response.getError();
                if (errors) {
                    if (errors[0] && errors[0].message) {
                        component.set("v.Error",errors[0].message);
                        console.log("Error message: " + 
                                 errors[0].message);
                    }
                } else {
                    console.log("Unknown error");
                }
            }
        });
        $A.enqueueAction(action);
    },
